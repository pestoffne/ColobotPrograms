public class TestSortedTreeIntInt extends EBoost {
	private SortedTreeIntInt tree = null;
	
	void setup() {
	}
	
	void cleanup() {
	}
	
	void test_1() {
		int v_0, v_111, v_222, v_333, v_444, v_555, v_666;
		//          33:333
		//  11:111          55:555
		// 0:0  22:222  44:444  66:666
		tree = new SortedTreeIntInt(7);
		
		for (int i = tree.size() - 1; i >= 0; --i) {
			tree.at_index(i).init(i * 11, i * 111);
		}
		
		v_0 = tree.get(0);
		v_111 = tree.get(11);
		v_222 = tree.get(22);
		v_333 = tree.get(33);
		v_444 = tree.get(44);
		v_555 = tree.get(55);
		v_666 = tree.get(66);
		
		assert_equal(v_0, 0);
		assert_equal(v_111, 111);
		assert_equal(v_222, 222);
		assert_equal(v_333, 333);
		assert_equal(v_444, 444);
		assert_equal(v_555, 555);
		assert_equal(v_666, 666);
	}
}

extern void Test_ListInt() {
	TestSortedTreeIntInt t();
	t.test_1();
	message("Tests passed", DisplayInfo);
}
