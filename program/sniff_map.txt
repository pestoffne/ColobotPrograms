extern void object::SniffMap() {
	point pos; int sign_x, sign_y;
	EBoost e(this);
	
	sign_x = 1;
	sign_y = 1;
	
	for (pos.y = -sign_y * 350; abs(pos.y) <= 350; pos.y += sign_y * 50) {
		for (pos.x = -sign_x * 350; abs(pos.x) <= 350; pos.x += sign_x * 50) {
			e.goto_o(flatspace(pos, 2.5, 0, 50));
			sniff();
			e.check_energy_loop();
			e.check_shield();
		}
		
		sign_x *= -1;
	}
	
	message("Scan is complete");
}
